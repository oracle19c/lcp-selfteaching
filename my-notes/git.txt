git clone                   -- 下载代码
git status                  -- 查看本地仓库的git状态
git add                     -- 将文件添加到暂存区，添加本地改动到下一次commit中
git commit                  -- 将暂存区的文件提交到本地仓库，提交本次commit到本地仓库,需要带上 -m "" 参数，参数内容为本次commit的描述
git push origin master      -- 将本地仓库的文件推送到远程仓库，提交本地仓库的commit list到GitHub的远程仓库中
